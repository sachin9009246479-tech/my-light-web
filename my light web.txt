  <script>
    // Simple JS for toggling 4 lights
    const lightsGrid = document.getElementById('lights');
    const lampElements = Array.from(lightsGrid.querySelectorAll('.lamp'));
    const onAllBtn = document.getElementById('onAll');
    const offAllBtn = document.getElementById('offAll');
    const randomBtn = document.getElementById('randomize');
    const resetBtn = document.getElementById('reset');

    function setLamp(lampEl, on){
      if(on){
        lampEl.classList.add('on');
        lampEl.setAttribute('aria-pressed','true');
        lampEl.querySelector('.status').textContent = 'ON';
      } else {
        lampEl.classList.remove('on');
        lampEl.setAttribute('aria-pressed','false');
        lampEl.querySelector('.status').textContent = 'OFF';
      }
    }

    // Initialize all OFF
    lampElements.forEach(l => setLamp(l, false));

    // Toggle buttons
    lightsGrid.addEventListener('click', e => {
      const toggle = e.target.closest('.toggle');
      const lampCard = e.target.closest('.lamp');
      if(toggle && lampCard){
        const isOn = lampCard.classList.contains('on');
        setLamp(lampCard, !isOn);
      } else if (lampCard && e.target.closest('.lamp') && !e.target.classList.contains('small')) {
        // clicking lamp area toggles too
        const isOn = lampCard.classList.contains('on');
        setLamp(lampCard, !isOn);
      }
    });

    // Keyboard: space/enter to toggle focused lamp (accessibility)
    lampElements.forEach(l=>{
      l.tabIndex = 0;
      l.addEventListener('keydown', ev=>{
        if(ev.key === 'Enter' || ev.key === ' '){
          ev.preventDefault();
          setLamp(l, !l.classList.contains('on'));
        }
      });
    });

    onAllBtn.addEventListener('click', ()=> lampElements.forEach(l => setLamp(l, true)));
    offAllBtn.addEventListener('click', ()=> lampElements.forEach(l => setLamp(l, false)));
    resetBtn.addEventListener('click', ()=> lampElements.forEach(l => setLamp(l, false)));
    randomBtn.addEventListener('click', ()=>{
      lampElements.forEach(l => setLamp(l, Math.random() > 0.5));
    });

    // Optional: expose control functions to console for tinkering
    window.lightsAPI = {
      set: (id, val) => {
        const el = lampElements.find(x => x.dataset.id === String(id));
        if(el) setLamp(el, !!val);
      },
      allOn: () => lampElements.forEach(l => setLamp(l, true)),
      allOff: () => lampElements.forEach(l => setLamp(l, false)),
      status: () => lampElements.map(l => ({ id: l.dataset.id, on: l.classList.contains('on') }))
    };
  </script>
